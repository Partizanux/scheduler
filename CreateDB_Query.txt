CREATE DATABASE mysitedb;

CREATE TABLE users( iduser MEDIUMINT NOT NULL AUTO_INCREMENT, firstName CHAR(20) NOT NULL, lastName CHAR(20) NOT NULL, login CHAR(15) NOT NULL, pass CHAR(15) NOT NULL, email CHAR(30) NOT NULL, PRIMARY KEY(iduser) );

CREATE TABLE tasks ( idtask MEDIUMINT NOT NULL AUTO_INCREMENT, date DATE NOT NULL, time TIME NOT NULL, task VARCHAR(300) NOT NULL, status BOOLEAN NOT NULL, notify TINYINT(1) NOT NULL, iduser MEDIUMINT NOT NULL, PRIMARY KEY(idtask), FOREIGN KEY(iduser) REFERENCES users (iduser) );
